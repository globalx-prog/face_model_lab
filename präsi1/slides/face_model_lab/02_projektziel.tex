\begin{frame}{Projektziel}
\begin{itemize}
    \item Datensatz WIDER FACE wird trainiert \newline --> Formatanpassung für Object Detection
    \item Mehrere Algorithmen werden trainiert und auf gleicher Datenbasis verglichen.
    \item Die Bewertung verbindet Recall, Precision, Latenz, Modellgrösse und Trainingsaufwand
    \item Das praktikabelste Modell wird in einer Video-Blur-Pipeline eingesetzt
\end{itemize}
\vspace{-0.9em}
\begin{mynote}[Pipeline]
Datensatz $\rightarrow$ Annotationen $\rightarrow$ Training $\rightarrow$ Evaluation $\rightarrow$ Video-Anonymisierung
\end{mynote}
\end{frame}
