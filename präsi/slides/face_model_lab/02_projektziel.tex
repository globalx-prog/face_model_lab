\begin{frame}{Projektziel}
\begin{itemize}
    \item WIDER FACE wird in ein trainierbares Format fuer Object Detection gebracht.
    \item Mehrere Detektoren werden trainiert und auf gleicher Datenbasis verglichen.
    \item Die Bewertung verbindet Recall, Precision, Latenz, Modellgroesse und Trainingsaufwand.
    \item Das praktikabelste Modell wird in einer Video-Blur-Pipeline eingesetzt.
\end{itemize}
\vspace{0.4em}
\begin{mynote}[Pipeline]
Datensatz $\rightarrow$ Annotationen $\rightarrow$ Training $\rightarrow$ Evaluation $\rightarrow$ Video-Anonymisierung
\end{mynote}
\end{frame}
